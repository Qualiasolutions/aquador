Create TypeScript script using tsx to verify indexes exist in Supabase after migration:

```typescript
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

const EXPECTED_INDEXES = [
  'idx_products_category',
  'idx_products_featured',
  'idx_products_active_category',
  'idx_blog_posts_status_slug',
  'idx_blog_posts_status_category',
  'idx_blog_posts_featured',
  'idx_orders_created_at',
  'idx_orders_customer_id',
];

async function verifyIndexes() {
  console.log('Checking indexes...');

  const { data, error } = await supabase.rpc('pg_indexes', {
    schemaname: 'public'
  });

  if (error) {
    console.error('Failed to query indexes:', error);
    process.exit(1);
  }

  const foundIndexes = EXPECTED_INDEXES.filter(idx =>
    data.some((row: any) => row.indexname === idx)
  );

  console.log(`Found ${foundIndexes.length}/${EXPECTED_INDEXES.length} indexes`);

  if (foundIndexes.length === EXPECTED_INDEXES.length) {
    console.log('✓ All indexes present');
    process.exit(0);
  } else {
    const missing = EXPECTED_INDEXES.filter(idx => !foundIndexes.includes(idx));
    console.error('✗ Missing indexes:', missing);
    process.exit(1);
  }
}

verifyIndexes();
```
