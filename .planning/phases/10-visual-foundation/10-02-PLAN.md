**src/components/ui/Section.tsx:**

```typescript
import React from 'react';
import { cn } from '@/lib/utils';

interface SectionProps {
  children: React.ReactNode;
  className?: string;
  size?: 'sm' | 'default' | 'lg';
  contained?: boolean;
  noPadding?: boolean;
  id?: string;
}

/**
 * Section component with consistent vertical rhythm
 *
 * @param size - Controls vertical padding (sm/default/lg)
 * @param contained - Constrains content to max-width container
 * @param noPadding - Removes all padding (for full-width backgrounds)
 */
export function Section({
  children,
  className,
  size = 'default',
  contained = false,
  noPadding = false,
  id,
}: SectionProps) {
  const sizeClasses = {
    sm: 'section-sm',
    default: 'section',
    lg: 'section-lg',
  };

  return (
    <section
      id={id}
      className={cn(
        !noPadding && sizeClasses[size],
        className
      )}
    >
      {contained ? (
        <div className="content-container">{children}</div>
      ) : (
        children
      )}
    </section>
  );
}
```

If cn utility doesn't exist, import from clsx:
```typescript

const cn = clsx;
```
