```typescript
/**
 * Generate blur data URL for placeholder
 * Uses tiny base64-encoded image for instant display
 */
export function getBlurDataURL(width = 10, height = 10): string {
  const svg = `
    <svg width="${width}" height="${height}" xmlns="http://www.w3.org/2000/svg">
      <defs>
        <linearGradient id="grad" x1="0%" y1="0%" x2="100%" y2="100%">
          <stop offset="0%" style="stop-color:#0a0a0a;stop-opacity:1" />
          <stop offset="50%" style="stop-color:#1a1a1a;stop-opacity:1" />
          <stop offset="100%" style="stop-color:#0a0a0a;stop-opacity:1" />
        </linearGradient>
      </defs>
      <rect width="${width}" height="${height}" fill="url(#grad)" />
    </svg>
  `;
  const base64 = Buffer.from(svg).toString('base64');
  return `data:image/svg+xml;base64,${base64}`;
}

/**
 * Generate responsive image sizes string
 * Optimizes bandwidth by serving appropriate size per viewport
 */
export function generateImageSizes(type: 'product' | 'hero' | 'thumbnail' | 'full'): string {
  const sizeMap = {
    // Product card images
    product: '(max-width: 640px) 100vw, (max-width: 1024px) 50vw, 33vw',
    // Hero/banner images
    hero: '100vw',
    // Small thumbnails
    thumbnail: '(max-width: 640px) 80px, 120px',
    // Full-width product detail
    full: '(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 600px',
  };
  return sizeMap[type];
}

/**
 * Normalize image URL from various sources
 * Handles Supabase storage, external URLs, and local paths
 */
export function getImageUrl(
  url: string | null | undefined,
  fallback = '/placeholder-product.svg'
): string {
  if (!url) return fallback;

  // Already a full URL
  if (url.startsWith('http://') || url.startsWith('https://')) {
    return url;
  }

  // Supabase storage path (needs full URL construction)
  if (url.includes('storage/v1/object/public/')) {
    return url;
  }

  // Local path
  if (url.startsWith('/')) {
    return url;
  }

  // Assume it's a relative path
  return `/${url}`;
}

/**
 * Generate srcset for manual responsive images
 * Use when Next.js Image component isn't suitable
 */
export function generateSrcSet(baseUrl: string, widths: number[]): string {
  return widths
    .map((width) => {
      const url = new URL(baseUrl, 'https://placeholder.com');
      url.searchParams.set('w', width.toString());
      return `${url.toString()} ${width}w`;
    })
    .join(', ');
}

/**
 * Common image dimensions for consistent aspect ratios
 */
export const IMAGE_DIMENSIONS = {
  productCard: { width: 400, height: 500 }, // 4:5 portrait
  productDetail: { width: 600, height: 750 }, // 4:5 portrait
  hero: { width: 1920, height: 800 }, // Wide banner
  thumbnail: { width: 120, height: 120 }, // Square
  blogFeatured: { width: 1200, height: 630 }, // OG image ratio
} as const;
```

```typescript
import Image from 'next/image';
import { getBlurDataURL, getImageUrl, generateImageSizes } from '@/lib/image-utils';
import { cn } from '@/lib/utils'; // or import clsx if utils doesn't exist

interface OptimizedImageProps {
  src: string | null | undefined;
  alt: string;
  width?: number;
  height?: number;
  fill?: boolean;
  sizes?: string;
  sizeType?: 'product' | 'hero' | 'thumbnail' | 'full';
  priority?: boolean;
  className?: string;
  objectFit?: 'cover' | 'contain' | 'fill';
  aspectRatio?: string;
  quality?: number;
}

/**
 * Optimized image component with automatic blur placeholders
 * Wraps Next.js Image with sensible defaults for luxury product photography
 */
export function OptimizedImage({
  src,
  alt,
  width,
  height,
  fill = false,
  sizes,
  sizeType = 'product',
  priority = false,
  className,
  objectFit = 'cover',
  aspectRatio,
  quality = 90,
}: OptimizedImageProps) {
  const imageUrl = getImageUrl(src);
  const imageSizes = sizes || generateImageSizes(sizeType);
  const blurDataURL = getBlurDataURL();

  // Container for aspect ratio when using fill
  if (fill) {
    return (
      <div
        className={cn('relative overflow-hidden', className)}
        style={aspectRatio ? { aspectRatio } : undefined}
      >
        <Image
          src={imageUrl}
          alt={alt}
          fill
          sizes={imageSizes}
          quality={quality}
          priority={priority}
          placeholder="blur"
          blurDataURL={blurDataURL}
          className={cn(
            objectFit === 'cover' && 'object-cover',
            objectFit === 'contain' && 'object-contain',
            objectFit === 'fill' && 'object-fill'
          )}
        />
      </div>
    );
  }

  // Fixed dimensions
  return (
    <Image
      src={imageUrl}
      alt={alt}
      width={width}
      height={height}
      sizes={imageSizes}
      quality={quality}
      priority={priority}
      placeholder="blur"
      blurDataURL={blurDataURL}
      className={className}
    />
  );
}
```

```typescript
import { OptimizedImage } from './OptimizedImage';
import { IMAGE_DIMENSIONS } from '@/lib/image-utils';
import { cn } from '@/lib/utils'; // or clsx

interface ProductImageProps {
  src: string | null | undefined;
  alt: string;
  variant?: 'card' | 'detail' | 'thumbnail';
  priority?: boolean;
  className?: string;
}

/**
 * Product image with consistent aspect ratios
 * Automatically applies correct dimensions based on variant
 */
export function ProductImage({
  src,
  alt,
  variant = 'card',
  priority = false,
  className,
}: ProductImageProps) {
  const dimensions = {
    card: IMAGE_DIMENSIONS.productCard,
    detail: IMAGE_DIMENSIONS.productDetail,
    thumbnail: IMAGE_DIMENSIONS.thumbnail,
  }[variant];

  const sizeType = {
    card: 'product' as const,
    detail: 'full' as const,
    thumbnail: 'thumbnail' as const,
  }[variant];

  return (
    <div className={cn('relative overflow-hidden bg-white', className)}>
      <OptimizedImage
        src={src}
        alt={alt}
        width={dimensions.width}
        height={dimensions.height}
        sizeType={sizeType}
        priority={priority}
        objectFit="cover"
        quality={variant === 'detail' ? 95 : 90}
        className="w-full h-auto"
      />
    </div>
  );
}
```
