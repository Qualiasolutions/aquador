Implementation details:
```typescript
'use client'
import { AnimatePresence, motion } from 'framer-motion'
import { usePathname } from 'next/navigation'
import { pageTransitionVariants, pageTransitionReducedMotion } from '@/lib/animations/page-transitions'
import { useReducedMotion } from '@/hooks/useReducedMotion' // if exists, else inline

export function PageTransition({ children }: { children: React.ReactNode }) {
  const pathname = usePathname()
  const prefersReducedMotion = useReducedMotion() // or inline: const [prefersReducedMotion, setPrefersReducedMotion] = useState(false); useEffect(...)

  const variants = prefersReducedMotion ? pageTransitionReducedMotion : pageTransitionVariants

  return (
    <AnimatePresence mode="wait" initial={false}>
      <motion.div
        key={pathname}
        variants={variants}
        initial="initial"
        animate="animate"
        exit="exit"
      >
        {children}
      </motion.div>
    </AnimatePresence>
  )
}
```

Critical considerations:
- initial={false} prevents animation on first load (faster perceived performance)
- mode="wait" ensures exit animation completes before enter starts
- Use pathname as key (not random ID) for consistent behavior
- Handle SSR safety (usePathname is safe in client components)
- Apply minimal styling to motion.div (no layout-affecting styles)

Error handling:
- Graceful fallback if usePathname fails (return children unwrapped)
- No console errors on navigation
