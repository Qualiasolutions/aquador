  <action>
Create `src/components/ui/ParallaxSection.tsx` drop-in wrapper component:
- Accept props: `children`, `speed` (number, default 0.5), `className` (string), `as` (element type, default 'div'), `disableOnMobile` (boolean, default true)
- Use `useParallax` hook with speed and disableOnMobile props
- Render motion component with ref attached
- Apply transform: `y: transform` (from hook) to create vertical parallax effect
- Support custom className for styling flexibility
- Use 'use client' directive (required for Framer Motion hooks)

Component structure:
```typescript
'use client';

import { motion } from 'framer-motion';
import { ReactNode, ElementType } from 'react';
import { useParallax } from '@/hooks/useParallax';

interface ParallaxSectionProps {
  children: ReactNode;
  speed?: number;
  className?: string;
  as?: ElementType;
  disableOnMobile?: boolean;
}

export function ParallaxSection({ children, speed = 0.5, className = '', as = 'div', disableOnMobile = true }: ParallaxSectionProps) {
  const { ref, transform, shouldAnimate } = useParallax({ speed, disableOnMobile });
  const Component = motion[as];

  return (
    <Component
      ref={ref}
      style={{ y: shouldAnimate ? transform : 0 }}
      className={className}
    >
      {children}
    </Component>
  );
}
```
