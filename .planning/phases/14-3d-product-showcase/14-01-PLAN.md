    Create src/components/3d/Scene.tsx:
    ```typescript
    'use client';

    import { Suspense } from 'react';
    import { Canvas } from '@react-three/fiber';
    import { OrbitControls } from '@react-three/drei';
    import { CAMERA_CONFIG, ORBIT_CONFIG } from '@/lib/three/config';

    function Loader() {
      return (
        <div className="flex items-center justify-center h-full">
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-gold-500" />
        </div>
      );
    }

    type SceneProps = {
      children: React.ReactNode;
      className?: string;
    };

    export function Scene({ children, className = 'w-full h-[600px]' }: SceneProps) {
      return (
        <div className={className}>
          <Canvas
            camera={CAMERA_CONFIG}
            gl={{ preserveDrawingBuffer: true }}
          >
            <Suspense fallback={null}>
              {children}
              <OrbitControls {...ORBIT_CONFIG} />
            </Suspense>
          </Canvas>
        </div>
      );
    }
    ```

    Key patterns from research:
    - 'use client' directive for Next.js App Router client-side rendering
    - Suspense wraps all 3D content for async GLTF loading
    - preserveDrawingBuffer allows canvas screenshots
    - OrbitControls with damping and polar angle limits (prevents upside-down views)
    - Uses Phase 10 design system color (border-gold-500) for loader
  </action>

<task type="auto">
  <name>Task 3: Implement realistic lighting with Environment and AccumulativeShadows</name>
  <files>
    src/components/3d/Lighting.tsx
  </files>
  <action>
    Create src/components/3d/Lighting.tsx with Environment + AccumulativeShadows:
    ```typescript
    import { Environment, AccumulativeShadows, RandomizedLight } from '@react-three/drei';
    import { LIGHTING_CONFIG } from '@/lib/three/config';

    type LightingProps = {
      simplified?: boolean;
    };

    export function Lighting({ simplified = false }: LightingProps) {
      if (simplified) {
        // Low-end devices: basic lighting, no shadows
        return (
          <>
            <ambientLight intensity={LIGHTING_CONFIG.ambientIntensity * 1.6} />
            <directionalLight
              position={LIGHTING_CONFIG.directionalPosition}
              intensity={LIGHTING_CONFIG.directionalIntensity * 0.5}
            />
          </>
        );
      }

      // High-end devices: full lighting with HDRI environment + realistic shadows
      return (
        <>
          {/* HDRI environment for reflections */}
          <Environment preset="city" background={false} />

          {/* Soft ambient + key light */}
          <ambientLight intensity={LIGHTING_CONFIG.ambientIntensity} />
          <directionalLight
            position={LIGHTING_CONFIG.directionalPosition}
            intensity={LIGHTING_CONFIG.directionalIntensity}
            castShadow
          />

          {/* Realistic ground shadows using temporal accumulation */}
          <AccumulativeShadows
            temporal
            frames={100}
            color="#9d4b4b"
            colorBlend={0.5}
            alphaTest={0.9}
            scale={20}
            position={[0, -1, 0]}
          >
            <RandomizedLight
              amount={8}
              radius={4}
              position={[5, 5, -10]}
            />
          </AccumulativeShadows>
        </>
      );
    }
    ```

    Research guidance:
    - Environment preset="city" provides HDRI-based realistic lighting without manual setup
    - AccumulativeShadows with temporal=true spreads shadow rendering across 100 frames for raycast-quality shadows at fraction of real-time cost
    - simplified prop will be used in Plan 04 for mobile performance optimization
    - background={false} keeps canvas transparent, letting Next.js page background show
  </action>
