    Create src/components/3d/CustomPerfumeBottle.tsx with procedural bottle geometry and note-based coloring:

    ```typescript
    'use client';

    import { useRef } from 'react';
    import { useFrame } from '@react-three/fiber';
    import * as THREE from 'three';

    type CustomPerfumeBottleProps = {
      topNoteColor?: string;
      heartNoteColor?: string;
      baseNoteColor?: string;
      autoRotate?: boolean;
    };

    export function CustomPerfumeBottle({
      topNoteColor = '#D4AF37',    // Default to Phase 10 gold
      heartNoteColor = '#D4AF37',
      baseNoteColor = '#D4AF37',
      autoRotate = true
    }: CustomPerfumeBottleProps) {
      const groupRef = useRef<THREE.Group>(null);

      // Auto-rotation using useFrame (mutates ref directly, never setState)
      useFrame((state, delta) => {
        if (groupRef.current && autoRotate) {
          groupRef.current.rotation.y += delta * 0.3;
        }
      });

      // Use heart note color for liquid (primary visual identity)
      const liquidColor = new THREE.Color(heartNoteColor);

      return (
        <group ref={groupRef}>
          {/* Glass bottle with transmission for realistic glass */}
          <mesh position={[0, 0, 0]}>
            <cylinderGeometry args={[0.5, 0.5, 2, 32]} />
            <meshPhysicalMaterial
              transmission={0.95}
              thickness={0.1}
              roughness={0.05}
              envMapIntensity={1}
              color="#ffffff"
            />
          </mesh>

          {/* Liquid inside bottle (heart note color) */}
          <mesh position={[0, -0.3, 0]}>
            <cylinderGeometry args={[0.45, 0.45, 1.4, 32]} />
            <meshStandardMaterial
              color={liquidColor}
              transparent
              opacity={0.8}
              metalness={0.1}
              roughness={0.2}
            />
          </mesh>

          {/* Gold cap (Phase 10 gold-500: #D4AF37) */}
          <mesh position={[0, 1.2, 0]}>
            <cylinderGeometry args={[0.3, 0.35, 0.4, 32]} />
            <meshStandardMaterial
              color="#D4AF37"
              metalness={0.9}
              roughness={0.1}
            />
          </mesh>

          {/* Base/stand (darker accent) */}
          <mesh position={[0, -1.2, 0]}>
            <cylinderGeometry args={[0.55, 0.6, 0.2, 32]} />
            <meshStandardMaterial
              color="#1a1a1a"
              metalness={0.3}
              roughness={0.7}
            />
          </mesh>
        </group>
      );
    }
    ```

    Design rationale:
    - Uses procedural geometry (no external model) for instant loading
    - meshPhysicalMaterial with transmission=0.95 creates realistic glass
    - heartNoteColor drives liquid color (matches user's main selection)
    - Auto-rotation showcases bottle from all angles
    - Gold cap uses Phase 10 OKLCH gold-500 (#D4AF37)
    - useFrame mutation pattern per research (never setState in render loop)

    Future enhancement: topNoteColor → gradient at top, baseNoteColor → gradient at bottom.
