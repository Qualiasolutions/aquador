  <action>
    Create src/hooks/useDeviceCapabilities.ts with mobile detection and memory/GPU heuristics:

    ```typescript
    import { useState, useEffect } from 'react';

    type DeviceCapabilities = {
      isMobile: boolean;
      isLowEnd: boolean;
      canHandle3D: boolean;
      recommendedDPR: number;
      memoryGB: number | null;
    };

    export function useDeviceCapabilities(): DeviceCapabilities {
      const [capabilities, setCapabilities] = useState<DeviceCapabilities>({
        isMobile: false,
        isLowEnd: false,
        canHandle3D: true,
        recommendedDPR: 1.5,
        memoryGB: null,
      });

      useEffect(() => {
        if (typeof window === 'undefined') return;

        // Mobile detection
        const isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent) ||
                        window.matchMedia('(max-width: 768px)').matches;

        // Memory detection (Chrome/Edge only, falls back to heuristics)
        // @ts-ignore - deviceMemory not in TypeScript DOM lib
        const memoryGB = (navigator.deviceMemory as number | undefined) || null;

        // GPU detection (experimental, not widely supported)
        const isLowEnd = memoryGB !== null && memoryGB < 4;

        // DPR recommendation
        const recommendedDPR = isMobile
          ? (isLowEnd ? 1 : 1.5)
          : window.devicePixelRatio > 2 ? 2 : 1.5;

        // Can handle 3D? (all devices can try, but warn on very low-end)
        const canHandle3D = !isLowEnd || memoryGB === null; // Optimistic if can't detect

        setCapabilities({
          isMobile,
          isLowEnd,
          canHandle3D,
          recommendedDPR,
          memoryGB,
        });
      }, []);

      return capabilities;
    }
    ```

    Detection strategy:
    - isMobile: User agent + viewport width check
    - memoryGB: Uses Device Memory API (Chrome/Edge only)
    - isLowEnd: <4GB RAM threshold
    - recommendedDPR: 1 for low-end, 1.5 for mobile, up to 2 for high-DPI desktop
    - canHandle3D: Optimistic unless confirmed low-end (user can still opt out)

    Research guidance: Mobile GPUs ~5x weaker, target <100 draw calls, disable shadows on mobile.
